using System;
using System.Collections;
using System.Linq;
using System.IO;
using System.Collections.Generic;

namespace Algoritms {
    class Program {
        static void Main(string[] args) {

            string s = "abcde";
        }
        public string solution(string s) {
            string answer = "";

            int slen = s.Length;

            if (s.Length % 2 == 0)
                answer = s.Substring(slen / 2 - 1, 2);
            else
                answer = s.Substring(slen / 2, 1);

            return answer;
        }
    }
}
