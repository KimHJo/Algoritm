using System;
using System.Collections;
using System.Linq;
using System.IO;
using System.Collections.Generic;

namespace Algoritms {
    class Program {
        static void Main(string[] args) {

        }

        public class Solution {
            public string solution(int n) {
                string answer = "";

                for (int i = 1; i <= n; i++) {
                    if (i % 2 == 1) {
                        answer += "수";
                    } else
                        answer += "박";
                }

                return answer;
            }
        }

    }

}
